    图形

    序言

    图形分为平面(2D)和立体(3D)

    平面

    p5js代码, 用顶点vertex( )绘制

    点

    function drawPoint(x,y)
{
    beginShape(POINTS);
        vertex(x, y);
    endShape();
}

    线

    function drawLine(x1,y1,x2,y2)
{
  beginShape();
    vertex(x1, y1);
    vertex(x2, y2);
  endShape();
}

    面(三角形)

    function drawTriangle(x1,y1,x2,y2,x3,y3)
{
  beginShape();
    vertex(x1, y1);
    vertex(x2, y2);
    vertex(x3, y3);
  endShape(CLOSE);
}

    面(正方形)

    function drawSquare(x, y, s) {
  beginShape();
  vertex(x, y);
  vertex(x + s, y);
  vertex(x + s, y + s);
  vertex(x, y + s);
  endShape(CLOSE);
}

    面(矩形)

    function drawRect(x, y, w, h) {
  beginShape();
  vertex(x, y);
  vertex(x + w, y);
  vertex(x + w, y + h);
  vertex(x, y + h);
  endShape(CLOSE);
}

    面(四角形)

    function drawQuad(x1, y1, x2,y2,x3,y3,x4,y4) {
  beginShape();
  vertex(x1, y1);
  vertex(x2, y2);
  vertex(x3, y3);
  vertex(x4, y4);
  endShape(CLOSE);
}

    面(正圆形)

    function drawCircle(x, y, r) {
  beginShape();
  for (let i = 0; i < 360; i++) {
    let angle = radians(i);
    let vertexX = x + cos(angle) * r;
    let vertexY = y + sin(angle) * r;
    vertex(vertexX, vertexY);
  }
  endShape(CLOSE);
}

    面(椭圆形)

    function drawEllipse(x, y, w, h) {
  beginShape();
  for (let i = 0; i < 360; i++) {
    let angle = radians(i);
    let vertexX = x + cos(angle) * w / 2;
    let vertexY = y + sin(angle) * h / 2;
    vertex(vertexX, vertexY);
  }
  endShape(CLOSE);
}
