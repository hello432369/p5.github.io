    function createPlayer() {       //复活初始值
  return { 
    x: 200,     //x坐标初始位置
    y: 300,     //y坐标初始位置
    vx: 0,      //x坐标速度
    vy: 0};     //y坐标速度
}

function updatePosition(entity) {       //运动方式
  entity.x += entity.vx;    //x坐标加速
  entity.y += entity.vy;    //y坐标加速
}

function applyGravity(entity) {     //受到的外力
  entity.vy += 0.15;    //y坐标向下重力运动
}

function applyJump(entity) {        //受到的控制力
  entity.vy = -5;       //y坐标向上蜻蜓点水
}   

function drawPlayer(entity) {       //外观
  square(entity.x, entity.y, 40);   //矩形
}

let player;

function setup() {
  createCanvas(800, 600);
  rectMode(CENTER);

  player = createPlayer();  //初始化
}

function draw() {

  updatePosition(player);   //移动
  applyGravity(player);   //y轴重力向下

  background(0);
  drawPlayer(player);       //展示游戏对象
}

function mousePressed() {
  applyJump(player);        //y轴轻功向上
}

    function createPlayer(x,y,vx,vy) {
  return {x ,y ,vx ,vy };
}

function updatePosition(entity) {
  entity.x += entity.vx;
  entity.y += entity.vy;
}

function applyGravity(entity) {
  entity.vy += 0.15;
}

function applyJump(entity) {
  entity.vy = -5;
}

function drawPlayer(mode,entity) {

  if(mode == 1)
  {
      noFill();
      strokeWeight(6);
      stroke(255);
      let time = frameCount * 0.2;
      let transformValue = 5 * sin(time);
      let width = 56 + transformValue;
      let height = 56 - transformValue;
      ellipse(entity.x, entity.y, width, height);
  }
  if(mode == 2)
  {
    push();

    translate(entity.x, entity.y);
    blendMode(ADD);
    strokeWeight(8);
    noFill();

    stroke(128);
    rect(0, 0, 80, 400);

    let x, y;
    let noiseTime = frameCount * 0.03;

    stroke(192, 0, 0);
    x = 50 * (-0.5 + noise(noiseTime));
    y = 50 * (-0.5 + noise(noiseTime + 100));
    rect(x, y, 80, 400);

    stroke(0, 128, 0);
    x = 50 * (-0.5 + noise(noiseTime + 200));
    y = 50 * (-0.5 + noise(noiseTime + 300));
    rect(x, y, 80, 400);

    stroke(0, 0, 128);
    x = 50 * (-0.5 + noise(noiseTime + 400));
    y = 50 * (-0.5 + noise(noiseTime + 500));
    rect(x, y, 80, 400);

    pop();
  }
}

let player; //玩家
let block;  //npc

function setup() {
  createCanvas(800, 600);
  rectMode(CENTER);

  player = createPlayer(200,300,0,0);       
  block = createPlayer(900,300,-2,0);
}

function draw() {
  updatePosition(player);
  updatePosition(block);    //npc坐标

  applyGravity(player);

  background(0);
  drawPlayer(1,player);
  drawPlayer(2,block);
}

function mousePressed() {
  applyJump(player);
