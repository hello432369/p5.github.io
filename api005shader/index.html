    颜色

    background()    设置背景颜色
fill()        填充颜色
stroke()        轮廓颜色
noFill()        无填充颜色
noStroke()    无轮廓颜色

colorMode()   颜色模式RGB, HSB
clear()       清除缓冲区的像素
erase()       擦除
noErase()       非擦除
    其他设置 
    color() 设置颜色
red()   获取红色
green() 获取绿色
blue()  获取蓝色
alpha() 获取透明度
hue()   获取HSB色相值
brightness()    获取HSB光的强弱
saturation()    获取HSB饱和度
lerpColor() 混合颜色
lightness() 获取HSL反射率
p5.Color    p5颜色

    precision mediump float;

attribute vec3 aPosition;
attribute vec2 aTexCoord;
attribute vec3 aNormal;

uniform mat4 uProjectionMatrix; 
uniform mat4 uModelViewMatrix; 

uniform float uFrameCount;

varying vec2 vTexCoord;
varying vec3 vNormal;


void main() {
  vec4 pv4 = vec4(aPosition, 1.0);

  vNormal = aNormal;
  vTexCoord = aTexCoord;

  pv4.x = sin(pv4.y*15.0  + uFrameCount * 0.09) * aNormal.x * 0.1;   //自定义：生成艺术的魅力，你永远都不知道偶然间会调出什么样的惊喜
  //pv4.x += sin(pv4.x*20.0  + uFrameCount * 0.1) * aNormal.x * 0.1; //原作者参数

  gl_Position = uProjectionMatrix * uModelViewMatrix * pv4;  //固定搭配, 3d投影到2d屏幕+坐标
}

    precision mediump float;

varying vec2 vTexCoord;

void main() {

}
